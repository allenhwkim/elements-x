How To Fire CustomEvent

CustomEvent polyfill and usage

This is polyfill for IE11

(function () {
  if (window.CustomEvent) return false;

  function CustomEvent(event, params) {
    params = params || { 
     bubbles: false, cancelable: false, detail: undefined 
    }; 
    var evt = document.createEvent( 'CustomEvent' );
    evt.initCustomEvent( 
      event, params.bubbles, params.cancelable, params.detail
    );
    return evt;
  }

  CustomEvent.prototype = window.Event.prototype;
  window.CustomEvent = CustomEvent;
})();

How to fire a custom event on an element

const el = document.getElementById('foo');
el.dispatchEvent(new CustomEvent('my-event', {
  bubbles: true, detail: new Date()
}));
