1. Install

$ npm install elements-x --save-dev

2. Import and Use it

import the component you need or import all, then use it in your html. To import all, use the following code: import { elementsX } 'elements-x'; elementsX.defineAll()

Javascript Example

import {XCalendar} 'elements-x'; XCalendar.define() const myDiv = document.getElementById('my'); myDiv.innerHTML = `<x-calendar>Hello</x-calendar>`; myDiv.addEventListener('click', function(event) { console.log(event) });

Angular Example

import { NgModule, Component, CUSTOM_ELEMENTS_SCHEMA } from '@angular/core'; import { BrowserModule } from '@angular/platform-browser'; import {XCalendar} 'elements-x'; XCalendar.define() @Component({ selector: 'my-app', template: '<x-calendar [select]="handleEvent($event)"></x-calendar>' }) class AppComponent {} @NgModule({ imports: [ BrowserModule ], declarations: [ AppComponent ], bootstrap: [ AppComponent ], schemas: [CUSTOM_ELEMENTS_SCHEMA] }) export class AppModule { handleEvent(event) { console.log(event); } } `

Vue Example

<template> <x-calendar @change="handleEvent"></x-calendar> </template> <script> import {XCalendar} 'elements-x'; XCalendar.define()> export default { name: "App", methods: { handleEvent: function(event) { console.log(event); } } }; </script>

ReactJs

import React from "react"; import {XCalendar} 'elements-x'; XCalendar.define(); export default function App() { componentDidMount() { ReactDOM.findDOMNode(this).addEventListener('click', this.handleEvent); } handleEvent(event) { console.log(event); } return ( <div> <x-calendar></x-calendar> </div> ); }
