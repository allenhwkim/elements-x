1. Install

$ npm install elements-x --save-dev

2. Import and Use it

import the component you need, then use it in your html.

Javascript Example

import {XButton} 'elements-x/button'; XButton.define() const myDiv = document.getElementById('my'); myDiv.innerHTML = `<x-button>Hello</x-button>`; myDiv.addEventListener('click', function(event) { console.log(event) });

Angular Example

import { NgModule, Component, CUSTOM_ELEMENTS_SCHEMA } from '@angular/core'; import { BrowserModule } from '@angular/platform-browser'; import {XButton} 'elements-x/button'; XButton.define() @Component({ selector: 'my-app', template: '<x-button>Hello</x-button>' }) class AppComponent {} @NgModule({ imports: [ BrowserModule ], declarations: [ AppComponent ], bootstrap: [ AppComponent ], schemas: [CUSTOM_ELEMENTS_SCHEMA] }) export class AppModule { handleEvent(event) { console.log(event); } } `

Vue Example

<template> <x-button v-on:click="handleEvent">Hello</x-button> </template> <script> import {XButton} 'elements-x/button'; XButton.define()> export default { name: "App", methods: { handleEvent: function(event) { console.log(event); } } }; </script>

ReactJs

import React from "react"; import {XButton} 'elements-x/button'; XButton.define() export default function App() { componentDidMount() { ReactDOM.findDOMNode(this).addEventListener('click', this.handleEvent); } handleEvent(event) { console.log(event); } return ( <div> <x-button>Hello</x-button> </div> ); }
