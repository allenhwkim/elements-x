3. Add prismjs to your app.module.ts


import { NgModule } from '@angular/core';
import { BrowserModule } from '@angular/platform-browser';
import { FormsModule } from '@angular/forms';

import { AppComponent } from './app.component';
import { **PrismComponent** } from './prism.component';

**import 'prismjs/components/prism-css';
import 'prismjs/components/prism-javascript';
import 'prismjs/components/prism-typescript';
import 'prismjs/components/prism-scss';**

@NgModule({
  imports:      [ BrowserModule, FormsModule ],
  declarations: [ AppComponent, **PrismComponent** ],
  bootstrap:    [ AppComponent ]
})
export class AppModule { }


4. Create prism.component.ts


import { Component, ElementRef, Input, AfterViewInit } from '[@angular/core](http://twitter.com/angular/core)';
**import Prism from 'prismjs';**

[@Component](http://twitter.com/Component)({
  selector: 'prism, [prism]',
  template: '<ng-content></ng-content>',
  styles: [`
    :host.dark { background: #333; color: #FFF;}`
  ]
})
export class PrismComponent implements AfterViewInit {
  [@Input](http://twitter.com/Input)() code: string;
  [@Input](http://twitter.com/Input)() language = 'javascript';

  constructor(private el: ElementRef) {}

  ngAfterViewInit() {
    const code = (this.code || this.el.nativeElement.innerText)
    const grammar =** Prism.languages[this.language];**
    const html = **Prism.highlight(code, grammar, this.language);**
    this.el.nativeElement.innerHTML = html;
  }
}
